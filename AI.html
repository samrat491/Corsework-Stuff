# --- STEP 1: PREPARE THE MODEL ---
# Using the existing 'df' from your previous cells
# We select the exact features present in your Diabetes.csv
feature_cols = ['Gender', 'AGE', 'Urea', 'Cr', 'HbA1c', 'Chol', 'TG', 'HDL', 'LDL', 'VLDL', 'BMI']

# We create a local copy to avoid "Double Encoding" errors if you run this cell twice
train_df = df.copy()

# Ensure data is clean (removing extra spaces from 'Y ', 'N ', etc.)
train_df['CLASS'] = train_df['CLASS'].astype(str).str.strip().str.upper()
train_df['Gender'] = train_df['Gender'].astype(str).str.strip().str.upper()

# Initialize Encoders
le_gender = LabelEncoder()
le_class = LabelEncoder()

# Encode features and target
train_df['Gender'] = le_gender.fit_transform(train_df['Gender'])
train_df['CLASS'] = le_class.fit_transform(train_df['CLASS'])

# Train the Decision Tree (dt)
X = train_df[feature_cols]
y = train_df['CLASS']
dt = DecisionTreeClassifier(random_state=42)
dt.fit(X, y)

print(" Model trained and calibrated using Diabetes.csv features.")

# --- STEP 2: INTERACTIVE PREDICTION SYSTEM ---
def interactive_prediction():
    print("\n" + " Diabetes Prediction System")
    print("-" * 35)
    
    try:
        # Collect User Inputs
        gender_raw = input("Enter Gender (M/F): ").strip().upper()
        age = float(input("Enter Age (years): "))
        urea = float(input("Enter Urea: "))
        cr = float(input("Enter Creatinine (Cr): "))
        hb = float(input("Enter HbA1c (%): "))
        chol = float(input("Enter Cholesterol: "))
        tg = float(input("Enter Triglycerides (TG): "))
        hdl = float(input("Enter HDL: "))
        ldl = float(input("Enter LDL: "))
        vldl = float(input("Enter VLDL: "))
        bmi = float(input("Enter BMI: "))

        # Convert input to DataFrame with Feature Names (Prevents the UserWarning)
        input_data = pd.DataFrame([[
            le_gender.transform([gender_raw])[0],
            age, urea, cr, hb, chol, tg, hdl, ldl, vldl, bmi
        ]], columns=feature_cols)

        # Make Prediction
        pred_idx = dt.predict(input_data)[0]
        result = le_class.inverse_transform([pred_idx])[0]

        # Final Formatted Output
        print("\n Prediction Result")
        print("-" * 25)
        
        mapping = {
            'Y': ' Result: Diabetic (Y)',
            'P': ' Result: Pre Diabetic (P)',
            'N': ' Result: Non Diabetic (N)'
        }
        
        print(mapping.get(result, f"Result: {result}"))
        print("-" * 25)

    except ValueError:
        print("\n Error: Please enter numeric values for medical data.")
    except Exception as e:
        print(f"\n Unexpected Error: {e}")

# Run the system
interactive_prediction()
